    ES6 Features¶

    1. let and const¶

    let x = 10;
const y = 20;

    2. Arrow Functions¶

    const greet = (name) => `Hello, ${name}`;

    3. Template Literals¶

    let name = "Alice";
console.log(`Hi, ${name}!`);

    4. Destructuring¶

    const user = { name: "Bob", age: 25 };
const { name, age } = user;

    5. Spread and Rest¶

    // Spread
let nums = [1, 2, 3];
let copy = [...nums];

// Rest
function sum(...args) {
  return args.reduce((a, b) => a + b);
}

    6. Default Parameters¶

    function greet(name = "Guest") {
  return `Hi, ${name}`;
}

    7. Classes¶

    class Person {
  constructor(name) {
    this.name = name;
  }

  greet() {
    return `Hi, I'm ${this.name}`;
  }
}
